	//this is the main method of the program
	public void cardValueUsers(Graphics art)
	{
		position5 = false;
		//the following happens if the user clicks stay
		if(stay)
		{
			//the dealer's second card flips
			back = false;
			timer ++;
			//after 200 milliseconds the round restarts
			if(timer == 200)
			{
				stay = false;
				playAgain(art);
			}

			

			if(firstStay)
			{
				//the seventh position card is turned over
				position7 = true;
				randomSuitandCard7(art);
				//calculates if the dealer should hit or not
				if(cardValue7 + cardValue8 < 17)
				{
					position5 = true;
					randomSuitandCard5(art);
					//calculates if the dealer should hit again or not
					if(cardValue7 + cardValue8 + cardValue5 < 17)
					{
						position6 = true;
					}
				}
				firstStay = false;
			}
			//prints out the dealer's card value
			firstbotCardValue = cardValue5 + cardValue7 + cardValue8 + cardValue6;
			//randomSuitandCard7(art);
			art.setColor(Color.yellow);
			art.drawString("Card Value: " +firstbotCardValue, 140, 240);
			art.setColor(Color.white);
			whoWins(art);
			score(art);
			}
	}
	//calculates if the dealer or user wins based on what cards are present
	public void whoWins(Graphics art)
	{
		if(firstbotCardValue >= 22)
		{
			if(userCardValue >= 22)
			{
				dealerWin = true;
			}
			if(userCardValue <= 21)
			{
				userWin = true;
			}
		}
		if(userCardValue > firstbotCardValue)
		{
			if(userCardValue >= 22)
			{
				dealerWin = true;
			}
			if(userCardValue <= 21)
			{
				userWin = true;
				
			}
		}
		if(firstbotCardValue > userCardValue)
		{
			if(firstbotCardValue <= 21)
			{
				dealerWin = true;
			}
			if(firstbotCardValue >= 22)
			{
				userWin = true;
			}
				
		}
		if(firstbotCardValue == userCardValue)
		{
			if(userCardValue <= 21)
			{
			art.setColor(Color.yellow);
			art.drawString("It's a tie! restarting", 400, 430);
			}
			else
			{
				dealerWin = true;
			}
		}
		
	}
		
		
	//updates the score each round after 199 milliseconds
	public void score(Graphics art)
	{
		if(userWin)
		{
			//userWin = false;
			art.setColor(Color.white);
			art.drawString("You Win!", 280, 320);
			if(timer == 199)
			{
				userRoundPoints ++;

				//reset(art);
			}
		}
		else
		{
			if(dealerWin)
			{

				art.setColor(Color.white);
				art.drawString("Dealer Wins!", 280, 320);
				if(timer == 199)
				{
					botRoundPoints ++;
					//dealerWin = false;
					//reset(art);
				}
			}
			else
			{
				if(tie)
				{
					art.setColor(Color.white);
					art.drawString("It's a tie! Restarting", 250, 320);
					//	reset(art);
				}
			}
		}
	}
